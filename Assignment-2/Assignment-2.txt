import pandas as pd

# Example of Quantitative Data (Numerical)
quantitative_data = {
    'Age': [25, 32, 47, 54, 23],
    'Income': [50000, 60000, 80000, 75000, 52000],
    'Height': [5.5, 6.0, 5.8, 5.6, 5.7]
}

# Convert to DataFrame
quantitative_df = pd.DataFrame(quantitative_data)
print("Quantitative Data:")
print(quantitative_df)

# Example of Qualitative Data (Categorical)
qualitative_data = {
    'Gender': ['Male', 'Female', 'Female', 'Male', 'Female'],
    'City': ['New York', 'Los Angeles', 'Chicago', 'Houston', 'Phoenix'],
    'Job': ['Engineer', 'Doctor', 'Teacher', 'Engineer', 'Nurse']
}

# Convert to DataFrame
qualitative_df = pd.DataFrame(qualitative_data)
print("\nQualitative Data:")
print(qualitative_df)

# Basic Statistical Summary for Quantitative Data
print("\nQuantitative Data Summary Statistics:")
print(quantitative_df.describe())

# Frequency count for Qualitative Data
print("\nQualitative Data Frequency Counts:")
print(qualitative_df['Gender'].value_counts())
print(qualitative_df['City'].value_counts())
